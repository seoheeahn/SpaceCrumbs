    <script>
        const questions = [
            { q: "새로운 상황에서의 반응", a: "모르는 사람들과 먼저 대화를 시작하는 편이다.", b: "새로운 사람들과 친해지는 데 시간이 걸린다." },
            { q: "하루 일과를 계획할 때", a: "하루 일정을 미리 정리하고 계획에 따라 움직인다.", b: "그때그때 상황에 따라 조정하는 편이다." }
        ];
        
        let currentQuestion = 0;
        let responses = [];
        
        function displayQuestion() {
            const container = document.getElementById("question-container");
            if (currentQuestion < questions.length) {
                container.innerHTML = `
                    <div class="question">
                        <h3>${questions[currentQuestion].q}</h3>
                        <button class="option-btn" onclick="selectOption(1)">${questions[currentQuestion].a}</button>
                        <button class="option-btn" onclick="selectOption(2)">중립</button>
                        <button class="option-btn" onclick="selectOption(3)">${questions[currentQuestion].b}</button>
                    </div>
                `;
                document.getElementById("next-btn").disabled = true;
            } else {
                calculateResult();
            }
        }
        
        function selectOption(value) {
            responses.push(value);
            document.getElementById("next-btn").disabled = false;
        }
        
        document.getElementById("next-btn").addEventListener("click", function() {
            currentQuestion++;
            displayQuestion();
        });
        
        function calculateResult() {
            document.querySelector(".test-container").style.display = "none";
            document.getElementById("result-container").style.display = "block";
            document.getElementById("result-desc").innerText = "테스트 완료! MBTI 분석 결과를 확인하세요.";
        }
        
        document.getElementById("restart-btn").addEventListener("click", function() {
            location.reload();
        });
        
        window.onload = displayQuestion;
    </script>
